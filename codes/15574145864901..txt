import java.util.Scanner;

/**
 * try
 */
public class Try {
   public static void main(String[] args) {
      Scanner scanner = new Scanner(System.in);
      int total = 100;

      while (true) {
         try {
            total -= scanner.nextInt();
         } catch (Exception e) {
            break;
         }
      }

      scanner.close();
      System.out.println(total);

   }
}
